      <span id="statusText">Not set</span>
    </div>

<script>
  const clockEl    = document.getElementById('clock');
  const dateEl     = document.getElementById('dateLabel');
  const alarmInput = document.getElementById('alarmTime');
  const toggleBtn  = document.getElementById('toggleBtn');
  const statusBadge= document.getElementById('statusBadge');
  const statusText = document.getElementById('statusText');
  const stopBtn    = document.getElementById('stopBtn');
  const card       = document.getElementById('card');

  let alarmOn = false, alarmTime = '', ringing = false;
  let audioCtx = null, ringerNodes = [];

  const DAYS = ['Sun','Mon','Tue','Wed','Thu','Fri','Sat'];
  const MONTHS = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'];

  function pad(n) { return String(n).padStart(2,'0'); }

  function tick() {
    const now = new Date();
    const hh = pad(now.getHours()), mm = pad(now.getMinutes()), ss = pad(now.getSeconds());
    clockEl.textContent = `${hh}:${mm}:${ss}`;
    dateEl.textContent  = `${DAYS[now.getDay()]}  ${MONTHS[now.getMonth()]} ${now.getDate()}`;

    if (alarmOn && !ringing && `${hh}:${mm}` === alarmTime && ss === '00') startRinging();
  }

  function startRinging() {
    ringing = true;
    statusBadge.className = 'status-badge ringing';
    statusText.textContent = 'Wake up!';
    stopBtn.style.display = 'block';
    card.classList.add('ringing');
    ring();
  }

  function ring() {
    if (!ringing) return;
    if (!audioCtx) audioCtx = new AudioContext();
    [0, 0.25, 0.5, 0.75].forEach(offset => {
      const osc = audioCtx.createOscillator();
      const gain = audioCtx.createGain();
      osc.connect(gain); gain.connect(audioCtx.destination);
      osc.frequency.value = 880; osc.type = 'sine';
      const t = audioCtx.currentTime + offset;
      gain.gain.setValueAtTime(0.4, t);
      gain.gain.exponentialRampToValueAtTime(0.001, t + 0.18);
      osc.start(t); osc.stop(t + 0.18);
      ringerNodes.push(osc);
    });
    setTimeout(() => { if (ringing) ring(); }, 1200);
  }

  function stopRinging() {
    ringing = false;
    ringerNodes.forEach(n => { try { n.stop(); } catch(_){} });
    ringerNodes = [];
    alarmOn = false;
    card.classList.remove('ringing');
    toggleBtn.textContent = 'Set';
    toggleBtn.className = 'btn-set';
    statusBadge.className = 'status-badge';
    statusText.textContent = 'Not set';
    stopBtn.style.display = 'none';
  }

  toggleBtn.addEventListener('click', () => {
    if (ringing) { stopRinging(); return; }
    if (alarmOn) {
      alarmOn = false;
      toggleBtn.textContent = 'Set';
      toggleBtn.className = 'btn-set';
      statusBadge.className = 'status-badge';
      statusText.textContent = 'Not set';
      return;
    }
    const val = alarmInput.value;
    if (!val) { statusText.textContent = 'Pick a time first'; return; }
    alarmTime = val; alarmOn = true;
    toggleBtn.textContent = 'Cancel';
    toggleBtn.className = 'btn-set cancel';
    statusBadge.className = 'status-badge on';
    statusText.textContent = `Set for ${val}`;
  });

  stopBtn.addEventListener('click', stopRinging);
  setInterval(tick, 1000);
  tick();
</script>
